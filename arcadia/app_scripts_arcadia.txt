function doPost(e) {
  const lock = LockService.getScriptLock();
  lock.waitLock(30000);

  try {
    _checkApiKey(e);

    if (!e || !e.postData || !e.postData.contents) {
      throw new Error('POST vacío.');
    }

    let payload;
    try {
      payload = JSON.parse(e.postData.contents);
    } catch (error) {
      throw new Error('JSON inválido en el cuerpo de la solicitud.');
    }

    const records = Array.isArray(payload) ? payload : [payload];
    if (!records.length) throw new Error('No hay registros para guardar.');

    const grupos = _agruparRegistrosPorDestino_(records);
    const detail = [];
    let totalRows = 0;

    grupos.forEach(g => {
      const { spreadsheet, sheet, headerInfo } = _abrirDestinoYPreparar_(g.destino);
      const rowsAdded = _guardarEnHoja_(sheet, headerInfo, g.records);
      totalRows += rowsAdded;
      detail.push({
        puntoVenta: g.destino.puntoVenta,
        spreadsheetId: spreadsheet.getId(),
        sheetName: sheet.getName(),
        rowsAdded: rowsAdded
      });
    });

    return _withCors(_json({
      result: 'success',
      rowsAdded: totalRows,
      detail: detail
    }), e);

  } catch (error) {
    Logger.log(error);
    return _withCors(_json({ result: 'error', error: String(error.message || error) }), e);
  } finally {
    lock.releaseLock();
  }
}

function doGet(e) {
  try {
    _checkApiKey(e);

    const p = e && e.parameter ? e.parameter : {};
    if (!p.action) return _withCors(_json({ result: 'error', error: 'Falta parámetro action.' }), e);

    const pvFilter = String(p.puntoVenta || '').trim();
    const pvFilterNorm = pvFilter ? _normText(pvFilter) : '';

    if (p.action === 'monthlySummary') {
      if (!p.date) return _withCors(_json({ result: 'error', error: 'Falta parámetro date.' }), e);
      if (!pvFilterNorm) return _withCors(_json({ result: 'error', error: 'Falta parámetro puntoVenta.' }), e);
      return _withCors(_json(leerResumenMensualHastaFecha(p.date, pvFilterNorm)), e);
    }

    if (p.action === 'conceptSummary') {
      if (!pvFilterNorm) return _withCors(_json({ result: 'error', error: 'Falta parámetro puntoVenta.' }), e);
      return _withCors(_json(leerResumenConceptosHojaCompleta(pvFilterNorm)), e);
    }

    if (p.action === 'pointSalesOnDate') {
      if (!p.date) return _withCors(_json({ result: 'error', error: 'Falta parámetro date.' }), e);
      return _withCors(_json(leerVentasPorPuntoEnFecha(p.date)), e);
    }

    if (!p.from || !p.to) return _withCors(_json({ result: 'error', error: 'Faltan parámetros from/to.' }), e);

    const tz = Session.getScriptTimeZone();
    const dFrom = _dayStart(p.from, tz);
    const dTo = _dayEnd(p.to, tz);
    if (isNaN(dFrom) || isNaN(dTo)) return _withCors(_json({ result: 'error', error: 'Formato de fecha inválido. Usa YYYY-MM-DD.' }), e);
    if (dFrom > dTo) return _withCors(_json({ result: 'error', error: 'from no puede ser mayor que to.' }), e);

    if (p.action === 'report') {
      return _withCors(_json(leerTotalesPorRango(p.from, p.to, pvFilterNorm)), e);
    }

    if (p.action === 'details') {
      return _withCors(_json(leerDetallesPorRango(p.from, p.to, pvFilterNorm)), e);
    }

    if (p.action === 'export') {
      const rows = leerTotalesPorRango(p.from, p.to, pvFilterNorm);
      const info = _crearArchivoExcelDesdeFilas(rows, `Reporte_${p.from}_a_${p.to}`);
      return _withCors(_json({ result: 'ok', fileId: info.fileId, downloadUrl: info.downloadUrl }), e);
    }

    return _withCors(_json({ result: 'error', error: 'Acción no soportada. Usa report, details, export, monthlySummary, conceptSummary o pointSalesOnDate.' }), e);
  } catch (error) {
    return _withCors(_json({ result: 'error', error: String(error.message || error) }), e);
  }
}


function leerResumenConceptosHojaCompleta(pvFilterNorm) {
  const puntoVenta = _buscarPuntoPorNorm_(pvFilterNorm);
  if (!puntoVenta) {
    throw new Error('El punto de venta solicitado no está configurado.');
  }

  const fuentes = _getFuentesDeDatos_(pvFilterNorm);
  const tz = Session.getScriptTimeZone();
  const acumulados = {};
  const orden = [];
  let fechaMin = '';
  let fechaMax = '';

  fuentes.forEach(fuente => {
    const spreadsheet = SpreadsheetApp.openById(fuente.spreadsheetId);
    const sheet = _getSheetByDestino_(spreadsheet, fuente.sheetName || '');
    if (!sheet) return;

    const range = sheet.getDataRange();
    const values = range.getValues();
    const displayValues = range.getDisplayValues();

    const headerRowIndex = encontrarFilaEncabezados(values, REQUIRED_HEADERS);
    if (headerRowIndex === -1) {
      throw new Error(`No se encontraron encabezados requeridos en ${sheet.getName()}: ${REQUIRED_HEADERS.join(', ')}`);
    }

    const headerMap = _crearMapaIndicesPorEncabezado_(values[headerRowIndex]);
    const idx = {
      fecha: headerMap[_normHeader_('fecha')],
      puntoVenta: headerMap[_normHeader_('puntoVenta')],
      tipo: headerMap[_normHeader_('tipo')],
      valor: headerMap[_normHeader_('valor')]
    };

    ['fecha', 'puntoVenta', 'tipo', 'valor'].forEach(function(campo) {
      if (idx[campo] == null || idx[campo] === -1) {
        throw new Error(`Falta columna ${campo} en ${sheet.getName()}.`);
      }
    });

    for (let r = headerRowIndex + 1; r < values.length; r++) {
      const rawRow = values[r] || [];
      const displayRow = displayValues[r] || [];

      const puntoRaw = displayRow[idx.puntoVenta] || rawRow[idx.puntoVenta] || '';
      const punto = String(puntoRaw).trim();
      if (!punto || _normText(punto) !== pvFilterNorm) continue;

      const fechaISO = _cellToIsoDate_(rawRow[idx.fecha], displayRow[idx.fecha], tz);
      if (fechaISO) {
        if (!fechaMin || fechaISO < fechaMin) fechaMin = fechaISO;
        if (!fechaMax || fechaISO > fechaMax) fechaMax = fechaISO;
      }

      const tipoRaw = displayRow[idx.tipo] || rawRow[idx.tipo] || '';
      const tipo = String(tipoRaw).trim();
      if (!tipo) continue;
      if (_normText(tipo).startsWith('total')) continue;

      const valorRaw = rawRow[idx.valor] !== '' && rawRow[idx.valor] != null
        ? rawRow[idx.valor]
        : displayRow[idx.valor];
      const valor = _toNumber(valorRaw);

      if (acumulados[tipo] == null) {
        acumulados[tipo] = 0;
        orden.push(tipo);
      }

      acumulados[tipo] += valor;
    }
  });

  const rangoTexto = fechaMin && fechaMax
    ? `Totales desde ${fechaMin} hasta ${fechaMax}`
    : 'Totales acumulados';

  return {
    result: 'ok',
    puntoVenta: puntoVenta,
    scope: 'all',
    excluyeConceptosDesde: 'total',
    fechaMin: fechaMin,
    fechaMax: fechaMax,
    rangoTexto: rangoTexto,
    conceptos: orden.map(tipo => ({
      tipo: tipo,
      valor: Number(acumulados[tipo] || 0)
    }))
  };
}

function leerVentasPorPuntoEnFecha(dateISO) {
  const value = String(dateISO || '').trim();
  if (!/^\d{4}-\d{2}-\d{2}$/.test(value)) {
    throw new Error('Formato de fecha inválido. Usa YYYY-MM-DD.');
  }

  const fuentes = _getFuentesDeDatos_('');
  const tz = Session.getScriptTimeZone();
  const ventasPorPunto = {};
  const tesoreriaPorPunto = {};
  const ordenPuntos = Object.keys(PUNTOS_VENTA_DESTINO);

  fuentes.forEach(fuente => {
    const spreadsheet = SpreadsheetApp.openById(fuente.spreadsheetId);
    const sheet = _getSheetByDestino_(spreadsheet, fuente.sheetName || '');
    if (!sheet) return;

    const range = sheet.getDataRange();
    const values = range.getValues();
    const displayValues = range.getDisplayValues();

    const headerRowIndex = encontrarFilaEncabezados(values, REQUIRED_HEADERS);
    if (headerRowIndex === -1) {
      throw new Error(`No se encontraron encabezados requeridos en ${sheet.getName()}: ${REQUIRED_HEADERS.join(', ')}`);
    }

    const headerMap = _crearMapaIndicesPorEncabezado_(values[headerRowIndex]);
    const idx = {
      fecha: headerMap[_normHeader_('fecha')],
      puntoVenta: headerMap[_normHeader_('puntoVenta')],
      tipo: headerMap[_normHeader_('tipo')],
      valor: headerMap[_normHeader_('valor')]
    };

    ['fecha', 'puntoVenta', 'tipo', 'valor'].forEach(function(campo) {
      if (idx[campo] == null || idx[campo] === -1) {
        throw new Error(`Falta columna ${campo} en ${sheet.getName()}.`);
      }
    });

    for (let r = headerRowIndex + 1; r < values.length; r++) {
      const rawRow = values[r] || [];
      const displayRow = displayValues[r] || [];

      const fechaISO = _cellToIsoDate_(rawRow[idx.fecha], displayRow[idx.fecha], tz);
      if (fechaISO !== value) continue;

      const puntoRaw = displayRow[idx.puntoVenta] || rawRow[idx.puntoVenta] || '';
      const punto = String(puntoRaw).trim();
      if (!punto) continue;

      const tipoRaw = displayRow[idx.tipo] || rawRow[idx.tipo] || '';
      const tipoNorm = _normText(String(tipoRaw).trim());
      if (tipoNorm !== TIPO_TOTAL_VENTAS_NORM && tipoNorm !== TIPO_TESORERIA_NORM) continue;

      const valorRaw = rawRow[idx.valor] !== '' && rawRow[idx.valor] != null
        ? rawRow[idx.valor]
        : displayRow[idx.valor];

      const valor = _toNumber(valorRaw);

      if (tipoNorm === TIPO_TOTAL_VENTAS_NORM) {
        ventasPorPunto[punto] = valor;
      }

      if (tipoNorm === TIPO_TESORERIA_NORM) {
        tesoreriaPorPunto[punto] = valor;
      }
    }
  });

  const puntosOrdenados = ordenPuntos
    .map(punto => ({
      punto: punto,
      valor: Number(ventasPorPunto[punto] || 0),
      tesoreria: Number(tesoreriaPorPunto[punto] || 0)
    }))
    .sort((a, b) => {
      const diff = Number(b.valor || 0) - Number(a.valor || 0);
      if (diff !== 0) return diff;
      return String(a.punto || '').localeCompare(String(b.punto || ''), 'es', { sensitivity: 'base' });
    });

  const totalGeneral = puntosOrdenados
    .reduce((sum, item) => sum + Number(item.valor || 0), 0);

  const totalTesoreria = puntosOrdenados
    .reduce((sum, item) => sum + Number(item.tesoreria || 0), 0);

  return {
    result: 'ok',
    fecha: value,
    rangoTexto: value,
    totalGeneral: Number(totalGeneral || 0),
    totalTesoreria: Number(totalTesoreria || 0),
    puntos: puntosOrdenados
  };
}

  }

  const fromISO = `${value.slice(0, 7)}-01`;
  const fuentes = _getFuentesDeDatos_(pvFilterNorm);
  const tz = Session.getScriptTimeZone();
  const ultimosTotalesPorFecha = {};

  fuentes.forEach(fuente => {
    const spreadsheet = SpreadsheetApp.openById(fuente.spreadsheetId);
    const sheet = _getSheetByDestino_(spreadsheet, fuente.sheetName || '');
    if (!sheet) return;

    const range = sheet.getDataRange();
    const values = range.getValues();
    const displayValues = range.getDisplayValues();

    const headerRowIndex = encontrarFilaEncabezados(values, REQUIRED_HEADERS);
    if (headerRowIndex === -1) {
      throw new Error(`No se encontraron encabezados requeridos en ${sheet.getName()}: ${REQUIRED_HEADERS.join(', ')}`);
    }

    const headerMap = _crearMapaIndicesPorEncabezado_(values[headerRowIndex]);
    const idx = {
      fecha: headerMap[_normHeader_('fecha')],
      puntoVenta: headerMap[_normHeader_('puntoVenta')],
      tipo: headerMap[_normHeader_('tipo')],
      valor: headerMap[_normHeader_('valor')]
    };

    ['fecha', 'puntoVenta', 'tipo', 'valor'].forEach(function(campo) {
      if (idx[campo] == null || idx[campo] === -1) {
        throw new Error(`Falta columna ${campo} en ${sheet.getName()}.`);
      }
    });

    for (let r = headerRowIndex + 1; r < values.length; r++) {
      const rawRow = values[r] || [];
      const displayRow = displayValues[r] || [];

      const fechaISO = _cellToIsoDate_(rawRow[idx.fecha], displayRow[idx.fecha], tz);
      if (!fechaISO) continue;
      if (fechaISO < fromISO || fechaISO > value) continue;

      const puntoRaw = displayRow[idx.puntoVenta] || rawRow[idx.puntoVenta] || '';
      const punto = String(puntoRaw).trim();
      if (!punto || _normText(punto) !== pvFilterNorm) continue;

        ? rawRow[idx.valor]
        : displayRow[idx.valor];
      ultimosTotalesPorFecha[fechaISO] = _toNumber(valorRaw);
    }
  });

  const ventasAcumuladas = Object.keys(ultimosTotalesPorFecha)
    .reduce((sum, fecha) => sum + _toNumber(ultimosTotalesPorFecha[fecha]), 0);

  return {
    result: 'ok',
    puntoVenta: puntoVenta,
    from: fromISO,
    to: value,
    monthLabel: _monthLabelEs_(value),
    ventasAcumuladas: Number(ventasAcumuladas || 0),
    diasConRegistro: Object.keys(ultimosTotalesPorFecha).length
  };
}

function _cellToIsoDate_(rawValue, displayValue, tz) {
  const display = String(displayValue == null ? '' : displayValue).trim();
  if (/^\d{4}-\d{2}-\d{2}$/.test(display)) {
    return display;
  }

  if (Object.prototype.toString.call(rawValue) === '[object Date]' && !isNaN(rawValue)) {
    return Utilities.formatDate(rawValue, tz, 'yyyy-MM-dd');
  }

  const coerced = _coerceFecha(rawValue);
  if (coerced) {
    return Utilities.formatDate(coerced, tz, 'yyyy-MM-dd');
  }

  if (/^\d{4}-\d{2}-\d{2}$/.test(display)) return display;
  return '';
}

function _monthLabelEs_(iso) {
  const parts = String(iso || '').split('-');
  if (parts.length !== 3) return '';
  const year = Number(parts[0]);
  const month = Number(parts[1]);
  const meses = [
    'Enero', 'Febrero', 'Marzo', 'Abril', 'Mayo', 'Junio',
    'Julio', 'Agosto', 'Septiembre', 'Octubre', 'Noviembre', 'Diciembre'
  ];
  if (!year || !month || month < 1 || month > 12) return '';
  return `${meses[month - 1]} ${year}`;
}


function _crearMapaIndicesPorEncabezado_(headerRow) {
  const map = {};
  (headerRow || []).forEach((value, index) => {
    const key = _normHeader_(value);
    if (key && map[key] == null) map[key] = index;
  });
  return map;
}

function _normHeader_(value) {
  return String(value == null ? '' : value)
    .trim()
    .replace(/\s+/g, '')
    .toLowerCase();
}

function leerTotalesPorRango(fromISO, toISO, pvFilterNorm) {
  const tz = Session.getScriptTimeZone();
  const dFrom = _dayStart(fromISO, tz);
  const dTo = _dayEnd(toISO, tz);
  const fuentes = _getFuentesDeDatos_(pvFilterNorm);
  const out = [];

  fuentes.forEach(fuente => {
    const info = _leerHojaYCabeceraDesdeFuente_(fuente);
    const data = info.data;
    const headerRowIndex = info.headerRowIndex;
    const idx = info.idx;

    for (let r = headerRowIndex + 1; r < data.length; r++) {
      const row = data[r];
      if (!row || row.length === 0 || row.every(v => v === '' || v === null)) continue;

      const f = _coerceFecha(row[idx.fecha]);
      if (!f) continue;
      if (f < dFrom || f > dTo) continue;

      const punto = String(row[idx.puntoVenta] || '').trim();
      if (pvFilterNorm && _normText(punto) !== pvFilterNorm) continue;

      const tipo = String(row[idx.tipo] || '').trim();
      if (!TOTALES_NORM.has(_normText(tipo))) continue;

      out.push({
        fecha: Utilities.formatDate(f, tz, 'yyyy-MM-dd'),
        puntoVenta: punto,
        tipo: tipo,
        valor: _toNumber(row[idx.valor])
      });
    }
  });

  out.sort((a, b) =>
    (a.fecha || '').localeCompare(b.fecha || '') ||
    (a.puntoVenta || '').localeCompare(b.puntoVenta || '') ||
    (a.tipo || '').localeCompare(b.tipo || '')
  );

  return out;
}

function _getFuentesDeDatos_(pvFilterNorm) {
  if (pvFilterNorm) {
    const punto = _buscarPuntoPorNorm_(pvFilterNorm);
    if (!punto) throw new Error('El punto de venta solicitado no está configurado.');
    const destino = _getDestinoPunto_(punto);
    return [{
      puntoVenta: punto,
      spreadsheetId: destino.spreadsheetId,
      sheetName: destino.sheetName || ''
    }];
  }

  const map = {};
  Object.keys(PUNTOS_VENTA_DESTINO).forEach(punto => {
    const destino = _getDestinoPunto_(punto);
    const key = `${destino.spreadsheetId}||${destino.sheetName || ''}`;
    if (!map[key]) {
      map[key] = {
        puntoVenta: punto,
        spreadsheetId: destino.spreadsheetId,
        sheetName: destino.sheetName || ''
      };
    }
  });

  return Object.keys(map).map(key => map[key]);
}

function _agruparRegistrosPorDestino_(records) {
  const groups = {};

  records.forEach((record, index) => {
    const puntoVenta = String(record && record.puntoVenta || '').trim();
    if (!puntoVenta) {
      throw new Error(`El registro ${index + 1} no trae puntoVenta.`);
    }

    const destino = _getDestinoPunto_(puntoVenta);
    const key = `${destino.spreadsheetId}||${destino.sheetName || ''}`;
    if (!groups[key]) {
      groups[key] = {
        destino: {
          puntoVenta: puntoVenta,
          spreadsheetId: destino.spreadsheetId,
          sheetName: destino.sheetName || ''
        },
        records: []
      };
    }
    groups[key].records.push(record);
  });

  return Object.keys(groups).map(key => groups[key]);
}

function _abrirDestinoYPreparar_(destino) {
  if (!destino || !destino.spreadsheetId) {
    throw new Error('Destino sin spreadsheetId.');
  }

  const spreadsheet = SpreadsheetApp.openById(destino.spreadsheetId);
  const sheet = _getSheetByDestino_(spreadsheet, destino.sheetName || '');
  if (!sheet) {
    throw new Error(`No se encontró la hoja destino para el Spreadsheet ID ${destino.spreadsheetId}.`);
  }

  const headerInfo = _leerHeaderInfo_(sheet, REQUIRED_HEADERS);
  return { spreadsheet, sheet, headerInfo };
}

function _getSheetByDestino_(spreadsheet, sheetName) {
  if (sheetName) {
    return spreadsheet.getSheetByName(sheetName) || null;
  }
  const sheets = spreadsheet.getSheets();
  return sheets && sheets.length ? sheets[SHEET_INDEX] : null;
}

function _leerHojaYCabeceraDesdeFuente_(fuente) {
  const spreadsheet = SpreadsheetApp.openById(fuente.spreadsheetId);
  const sheet = _getSheetByDestino_(spreadsheet, fuente.sheetName || '');
  if (!sheet) throw new Error(`No se encontró hoja en ${fuente.spreadsheetId}.`);

  const data = sheet.getDataRange().getValues();
  const headerRowIndex = encontrarFilaEncabezados(data, REQUIRED_HEADERS);
  if (headerRowIndex === -1) {
    throw new Error(`No se encontraron encabezados requeridos en ${sheet.getName()}: ${REQUIRED_HEADERS.join(', ')}`);
  }

  const headerMap = _crearMapaIndicesPorEncabezado_(data[headerRowIndex]);
  const idx = Object.fromEntries(REQUIRED_HEADERS.map(h => [h, headerMap[_normHeader_(h)]]));
  REQUIRED_HEADERS.forEach(h => {
    if (idx[h] == null || idx[h] === -1) throw new Error(`Falta columna ${h} en ${sheet.getName()}.`);
  });

  return { data, headerRowIndex, idx };
}

function _guardarEnHoja_(sheet, headerInfo, records) {
  if (!records || !records.length) return 0;

  const headerRowIndex = headerInfo.headerRowIndex;
  const header = headerInfo.header;
  const colIdx = headerInfo.colIdx;
  const optNumIdx = headerInfo.optNumIdx;

  const maxWriteIndex = Math.max.apply(null, Object.values(colIdx).concat(Object.values(optNumIdx)));
  const lastColToWrite = maxWriteIndex + 1;

  const rows = records.map((rec, index) => {
    REQUIRED_HEADERS.forEach(h => {
      if (!(h in rec)) throw new Error(`Falta el campo requerido ${h} en el registro ${index + 1}.`);
    });

    let valor = rec.valor;
    if (typeof valor === 'string') {
      valor = valor.replace(/\s/g, '').replace(/\./g, '').replace(',', '.');
      valor = parseFloat(valor);
    }
    if (typeof valor !== 'number' || !isFinite(valor)) valor = 0;

    const row = new Array(lastColToWrite).fill('');
    row[colIdx.fecha] = rec.fecha;
    row[colIdx.puntoVenta] = rec.puntoVenta;
    row[colIdx.tipo] = rec.tipo;
    row[colIdx.tercero] = rec.tercero;
    row[colIdx.detalle] = _capitalizarPalabras_(rec.detalle);
    row[colIdx.valor] = Number(valor) || 0;

    Object.keys(optNumIdx).forEach(k => {
      let n = rec[k];
      if (typeof n === 'string') {
        n = n.replace(/\s/g, '').replace(/\./g, '').replace(',', '.');
        n = parseFloat(n);
      }
      if (typeof n !== 'number' || !isFinite(n)) n = 0;
      row[optNumIdx[k]] = Number(n) || 0;
    });

    return row;
  });

  const startRow = Math.max(sheet.getLastRow() + 1, headerRowIndex + 2);
  sheet.getRange(startRow, 1, rows.length, rows[0].length).setValues(rows);
  sheet.getRange(startRow, colIdx.valor + 1, rows.length, 1).setNumberFormat('#,##0');

  Object.keys(optNumIdx).forEach(k => {
    sheet.getRange(startRow, optNumIdx[k] + 1, rows.length, 1).setNumberFormat('#,##0');
  });

  const maxColsHoja = sheet.getMaxColumns();
  const highlightStartCol = FORMAT_COL_START;
  const highlightEndCol = Math.min(FORMAT_COL_END, maxColsHoja);
  const highlightWidth = Math.max(0, highlightEndCol - highlightStartCol + 1);
  const normalizeWidth = Math.min(Math.max(header.length || 0, lastColToWrite, 10), maxColsHoja);

  sheet.getRange(startRow, 1, rows.length, normalizeWidth).setFontWeight('normal');
  sheet.getRange(startRow, 1, rows.length, normalizeWidth).setBackground(null);

  for (let i = 0; i < records.length; i++) {
    const tipoRaw = String(records[i] && records[i].tipo || '').trim();
    const tipoNorm = _normText(tipoRaw);
    if (!TOTALES_NORM.has(tipoNorm) || highlightWidth <= 0) continue;

    const rg = sheet.getRange(startRow + i, highlightStartCol, 1, highlightWidth);
    rg.setFontWeight('bold');

    if (tipoNorm === TIPO_TESORERIA_NORM) {
      rg.setBackgroundRGB(COLOR_TESORERIA_RGB.r, COLOR_TESORERIA_RGB.g, COLOR_TESORERIA_RGB.b);
    }
  }

  return rows.length;
}

function _leerHeaderInfo_(sheet, requiredHeaders) {
  const data = sheet.getDataRange().getValues();
  const headerRowIndex = encontrarFilaEncabezados(data, requiredHeaders);
  if (headerRowIndex === -1) {
    throw new Error(`No se encontraron los encabezados requeridos en ${sheet.getName()}: ${requiredHeaders.join(', ')}`);
  }

  const header = data[headerRowIndex].map(v => String(v).trim());
  const headerMap = _crearMapaIndicesPorEncabezado_(data[headerRowIndex]);
  const colIdx = Object.fromEntries(requiredHeaders.map(h => [h, headerMap[_normHeader_(h)]]));
  requiredHeaders.forEach(h => {
    if (colIdx[h] == null || colIdx[h] === -1) throw new Error(`Falta columna ${h} en ${sheet.getName()}.`);
  });

  const optNumIdx = {};
  OPTIONAL_NUM_HEADERS.forEach(h => {
    const idx = headerMap[_normHeader_(h)];
    if (idx != null && idx !== -1) optNumIdx[h] = idx;
  });

  return { headerRowIndex, header, colIdx, optNumIdx };
}

function encontrarFilaEncabezados(data, requiredHeaders) {
  const requiredNorm = (requiredHeaders || []).map(_normHeader_);
  for (let i = 0; i < data.length; i++) {
    const rowNorm = (data[i] || []).map(_normHeader_);
    const ok = requiredNorm.every(h => rowNorm.indexOf(h) !== -1);
    if (ok) return i;
  }
  return -1;
}

function _buscarPuntoPorNorm_(pvFilterNorm) {
  const puntos = Object.keys(PUNTOS_VENTA_DESTINO);
  for (let i = 0; i < puntos.length; i++) {
    if (_normText(puntos[i]) === pvFilterNorm) return puntos[i];
  }
  return '';
}

function _getDestinoPunto_(puntoVenta) {
  const destino = PUNTOS_VENTA_DESTINO[String(puntoVenta || '').trim()];
  if (!destino) throw new Error(`El punto de venta ${puntoVenta} no está configurado.`);
  if (!destino.spreadsheetId || String(destino.spreadsheetId).indexOf('REEMPLAZAR_') === 0) {
    throw new Error(`Debes configurar el spreadsheetId del punto de venta ${puntoVenta}.`);
  }
  return destino;
}

function _crearArchivoExcelDesdeFilas(rows, nombreBase) {
  const headers = ['fecha', 'puntoVenta', 'tipo', 'valor'];
  const ss = SpreadsheetApp.create(nombreBase || 'Reporte');
  const sh = ss.getActiveSheet();
  sh.setName('Reporte');

  const matrix = [headers].concat(rows.map(r => [r.fecha, r.puntoVenta, r.tipo, r.valor]));
  sh.getRange(1, 1, matrix.length, headers.length).setValues(matrix);
  sh.getRange(1, 1, 1, headers.length).setFontWeight('bold');
  sh.autoResizeColumns(1, headers.length);

  const fileId = ss.getId();
  const downloadUrl = `https://docs.google.com/spreadsheets/d/${fileId}/export?format=xlsx`;
  return { fileId: fileId, downloadUrl: downloadUrl };
}

function _normText(value) {
  return String(value || '')
    .toLowerCase()
    .normalize('NFD')
    .replace(/[\u0300-\u036f]/g, '')
    .replace(/[^a-z0-9]+/g, ' ')
    .replace(/\s+/g, ' ')
    .trim();
}

function _capitalizarPalabras_(value) {
  return String(value || '')
    .trim()
    .toLowerCase()
    .replace(/\s+/g, ' ')
    .replace(/(^|[\s\-\/\(])([a-záéíóúñü])/g, function(_, separador, letra) {
      return separador + letra.toUpperCase();
    });
}

function _toNumber(value) {
  if (typeof value === 'number') return isFinite(value) ? value : 0;
  let s = String(value == null ? '' : value).trim();
  if (!s) return 0;
  s = s.replace(/\s/g, '').replace(/\./g, '').replace(',', '.');
  const n = parseFloat(s);
  return isFinite(n) ? n : 0;
}

function _coerceFecha(value) {
  if (Object.prototype.toString.call(value) === '[object Date]' && !isNaN(value)) {
    return value;
  }
  const s = String(value == null ? '' : value).trim();
  if (!s) return null;
  if (/^\d{4}-\d{2}-\d{2}$/.test(s)) {
    const d = new Date(`${s}T00:00:00`);
    return isNaN(d) ? null : d;
  }
  const d = new Date(s);
  return isNaN(d) ? null : d;
}

function _dayStart(iso, tz) {
  const parts = String(iso || '').split('-');
  if (parts.length !== 3) return new Date('invalid');
  const d = new Date(Number(parts[0]), Number(parts[1]) - 1, Number(parts[2]), 0, 0, 0, 0);
  return d;
}

function _dayEnd(iso, tz) {
  const parts = String(iso || '').split('-');
  if (parts.length !== 3) return new Date('invalid');
  const d = new Date(Number(parts[0]), Number(parts[1]) - 1, Number(parts[2]), 23, 59, 59, 999);
  return d;
}
